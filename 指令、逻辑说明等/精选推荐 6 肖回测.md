  * 精选推荐 6 肖回测（用于 #zodiacFinalNum 点击弹窗）
   * 算法：对每一期回测目标，取其前 12 期窗口统计生肖频次，
   *       选 Top6 作为"内幕六肖"预测；与实际开出生肖比对判定命中。
   * @param {Array} historyData - 历史数据（[0] 最新，[1] 次新，…）
   * @param {number} testCount - 回测期数（默认 20，上限 30）
   * @returns {Object|null} 回测汇总
   */
  runFinalZodiacBacktest: function(historyData, testCount) {
    if (!historyData || historyData.length < 14) return null;
    testCount = Math.min(testCount || 20, 30);
    var allZodiacs = ZodiacPrediction.ZODIAC_ORDER;
    var results = [];

    for (var offset = 0; offset < testCount; offset++) {
      var targetItem = historyData[offset];
      if (!targetItem) break;
      // 取目标期之前 12 期作为"统计窗口"
      var windowStart = offset + 1;
      var windowEnd = Math.min(windowStart + 12, historyData.length);
      if (windowEnd - windowStart < 6) break; // 窗口不足 6 期直接放弃
      var window = historyData.slice(windowStart, windowEnd);

      // 统计窗口内每个生肖出现次数
      var zodCount = {};
      allZodiacs.forEach(function(z) { zodCount[z] = 0; });
      window.forEach(function(item) {
        var s = Utils.SpecialCalculator.getSpecial(item);
        if (s && s.zod && zodCount.hasOwnProperty(s.zod)) {
          zodCount[s.zod]++;
        }
      });

      // 按频次降序取 Top6（平局按 ZODIAC_ORDER 顺序稳定）
      var top6 = allZodiacs.slice().sort(function(a, b) {
        if (zodCount[b] !== zodCount[a]) return zodCount[b] - zodCount[a];
        return allZodiacs.indexOf(a) - allZodiacs.indexOf(b);
      }).slice(0, 6);

      // 实际开出的生肖与特码
      var actualSpecial = Utils.SpecialCalculator.getSpecial(targetItem);
      var actualZod = actualSpecial.zod || '-';
      var actualNum = actualSpecial.te || 0;
      var isHit = top6.indexOf(actualZod) !== -1;

      results.push({
        expect: targetItem.expect,
        top6Zodiacs: top6,
        actualZodiac: actualZod,
        actualNumber: actualNum,
        isHit: isHit
      });
    }

    if (!results.length) return null;

    var hitCount = results.filter(function(r) { return r.isHit; }).length;
    var hitRate = Math.round((hitCount / results.length) * 100);
    var recentResults = results.slice(0, 12);
    var recentHits = recentResults.filter(function(r) { return r.isHit; }).length;
    var recentHitRate = recentResults.length > 0 ? Math.round((recentHits / recentResults.length) * 100) : 0;
    var currentStreak = 0;
    for (var i = 0; i < recentResults.length; i++) {
      if (recentResults[i].isHit) currentStreak++;
      else break;
    }

    return {
      totalTests: results.length,
      totalHits: hitCount,
      totalHitRate: hitRate,
      recentTests: recentResults.length,
      recentHits: recentHits,
      recentHitRate: recentHitRate,
      currentStreak: currentStreak,
      details: recentResults
    };
  },
