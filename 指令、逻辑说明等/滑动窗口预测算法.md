# 滑动窗口预测算法 · 前端项目设计方案

---

## 一、项目概述

### 1.1 项目名称
**生肖预测系统**（Shengxiao Predictor）

### 1.2 项目目标
基于12/24/36期三窗联合判定算法，自动计算每期6个候选生肖，实现高准确率预测。

### 1.3 技术栈
- **前端框架**：React 18 + TypeScript
- **UI 组件库**：Ant Design 5
- **状态管理**：Zustand
- **图表库**：ECharts
- **构建工具**：Vite
- **样式方案**：Tailwind CSS + Ant Design

### 2.3 窗口计算模块

```typescript
// src/algorithm/window.ts

import { Shengxiao, Period, WindowData } from './types';

export function calculateWindows(
  data: Period[], 
  currentIndex: number
): WindowData {
  const window12 = new Map<Shengxiao, number>();
  const window11 = new Map<Shengxiao, number>();
  const window24 = new Map<Shengxiao, number>();
  const window36 = new Map<Shengxiao, number>();
  
  // 初始化
  const allShengxiao: Shengxiao[] = [
    '鼠', '牛', '虎', '兔', '龙', '蛇',
    '马', '羊', '猴', '鸡', '狗', '猪'
  ];
  allShengxiao.forEach(s => {
    window12.set(s, 0);
    window11.set(s, 0);
    window24.set(s, 0);
    window36.set(s, 0);
  });
  
  // 计算各窗口
  for (let i = 0; i < 36 && i <= currentIndex; i++) {
    const period = data[currentIndex - i];
    if (!period) break;
    
    const sx = period.shengxiao;
    if (i < 12) window12.set(sx, (window12.get(sx) || 0) + 1);
    if (i < 11) window11.set(sx, (window11.get(sx) || 0) + 1);
    if (i < 24) window24.set(sx, (window24.get(sx) || 0) + 1);
    if (i < 36) window36.set(sx, (window36.get(sx) || 0) + 1);
  }
  
  return { window12, window11, window24, window36 };
}
```

### 2.5 核心评分规则

```typescript
// src/algorithm/rules.ts

import { Shengxiao, CandidateScore, WindowData } from './types';
import { getZone12, getZone24, getZone36 } from './zones';

export function calculateScore(
  shengxiao: Shengxiao,
  windows: WindowData,
  lastPeriodIndex: number,
  totalPeriods: number
): CandidateScore {
  const w12 = windows.window12.get(shengxiao) || 0;
  const w11 = windows.window11.get(shengxiao) || 0;
  const w24 = windows.window24.get(shengxiao) || 0;
  const w36 = windows.window36.get(shengxiao) || 0;
  
  let score = 0;
  const signals: string[] = [];
  const reasons: string[] = [];
  
  // 规则1：24/36期双过热(5/7) - 最强信号
  if (w24 === 5 && w36 === 7) {
    score += 100;
    signals.push('24/36期双过热(5/7)');
    reasons.push('24/36期双过热是最强信号');
  }
  // 规则2：24/36期双过热(5/6)
  else if (w24 >= 5 && w36 >= 6) {
    score += 90;
    signals.push('24/36期双过热');
    reasons.push('24/36期双过热是极强信号');
  }
  // 规则3：三窗热号(2/4/5+)
  else if (w12 === 2 && w24 === 4 && w36 >= 5) {
    score += 80;
    signals.push('三窗热号');
    reasons.push('三窗热号是强信号');
  }
  // 规则4：24/36期双热号(4/6)
  else if (w24 === 4 && w36 === 6) {
    score += 70;
    signals.push('24/36期双热号(4/6)');
    reasons.push('24/36期双热号是强信号');
  }
  // 规则5：24/36期热号(4/5)
  else if (w24 === 4 && w36 === 5) {
    score += 60;
    signals.push('24/36期热号(4/5)');
    reasons.push('24/36期热号是中强信号');
  }
  // 规则6：36期热号(6)
  else if (w36 === 6) {
    score += 55;
    signals.push('36期热号(6)');
    reasons.push('36期热号(6)是中强信号');
  }
  // 规则7：24/36期活跃(3/4)
  else if (w24 === 3 && w36 === 4) {
    score += 45;
    signals.push('24/36期活跃(3/4)');
    reasons.push('24/36期活跃是中等信号');
  }
  // 规则8：36期活跃(4)
  else if (w36 === 4) {
    score += 40;
    signals.push('36期活跃(4)');
    reasons.push('36期活跃(4)是中等信号');
  }
  // 规则9：36期活跃(3)
  else if (w36 === 3) {
    score += 30;
    signals.push('36期活跃(3)');
    reasons.push('36期活跃(3)是中等信号');
  }
  // 规则10：24/36期双冷(1/2) - 必出信号
  else if (w24 === 1 && w36 === 2) {
    score += 50;
    signals.push('24/36期双冷+超长遗漏');
    reasons.push('24/36期双冷+超长遗漏是必出信号');
  }
  // 规则11：2.7规则触发（穿插+活跃+热号）
  else if (w12 === 1 && w24 === 3 && w36 >= 5) {
    score += 65;
    signals.push('2.7规则触发');
    reasons.push('2.7规则触发（穿插+活跃+热号）');
  }
  // 规则12：2.7规则触发（穿插+热号+热号）
  else if (w12 === 1 && w24 === 4 && w36 >= 4) {
    score += 65;
    signals.push('2.7规则触发');
    reasons.push('2.7规则触发（穿插+热号+热号）');
  }
  
  // 冷补不重复扣分
  const periodsSinceLast = getPeriodsSinceLast(shengxiao, lastPeriodIndex, totalPeriods);
  if (periodsSinceLast <= 2) {
    // 刚开过，检查是否有强信号覆盖
    if (!signals.some(s => s.includes('双过热'))) {
      score -= 20;
      reasons.push(`冷补不重复：${periodsSinceLast}期前刚开过`);
    }
  } else if (periodsSinceLast >= 6 && periodsSinceLast <= 14) {
    // 刚过冷补期
    if (score > 0) {
      score += 10;
      reasons.push(`已过冷补期：${periodsSinceLast}期前`);
    }
  } else if (periodsSinceLast >= 15) {
    // 超长遗漏
    if (w24 <= 2 && w36 <= 2) {
      score += 30;
      signals.push('超长遗漏');
      reasons.push(`超长遗漏：${periodsSinceLast}期未出`);
    }
  }
  
  return {
    shengxiao,
    score,
    reason: reasons.join('；'),
    signals,
    window12: w12,
    window11: w11,
    window24: w24,
    window36: w36
  };
}

function getPeriodsSinceLast(
  shengxiao: Shengxiao,
  currentIndex: number,
  totalPeriods: number
): number {
  // 计算该生肖距离最近一次出现的期数
  for (let i = 1; i <= totalPeriods; i++) {
    if (currentIndex - i < 0) break;
    // 这里需要传入历史数据来查询
    return i; // 简化实现
  }
  return totalPeriods;
}
```

### 2.6 预测主逻辑

```typescript
// src/algorithm/predict.ts

import { Period, PredictionResult, CandidateScore, Shengxiao } from './types';
import { calculateWindows } from './window';
import { calculateScore } from './rules';

export function predict(
  historyData: Period[],
  currentPeriod: number
): PredictionResult {
  // 找到当前期在历史数据中的位置
  const currentIndex = historyData.findIndex(p => p.period === currentPeriod - 1);
  if (currentIndex === -1) {
    throw new Error('历史数据不足');
  }
  
  // 计算窗口
  const windows = calculateWindows(historyData, currentIndex);
  
  // 计算所有生肖的评分
  const allShengxiao: Shengxiao[] = [
    '鼠', '牛', '虎', '兔', '龙', '蛇',
    '马', '羊', '猴', '鸡', '狗', '猪'
  ];
  
  const scores: CandidateScore[] = allShengxiao.map(sx => 
    calculateScore(sx, windows, currentIndex, historyData.length)
  );
  
  // 按评分排序，取前6
  const sortedScores = [...scores].sort((a, b) => b.score - a.score);
  const top6 = sortedScores.slice(0, 6);
  
  return {
    period: currentPeriod,
    candidates: top6.map(s => s.shengxiao),
    scores: top6,
    algorithm: 'V5.3',
    timestamp: Date.now()
  };
}
```

### 2.7 验证模块

```typescript
// src/algorithm/validate.ts

import { Period, PredictionResult } from './types';

export interface ValidationResult {
  period: number;
  actual: string;
  predicted: string[];
  hit: boolean;
  rank: number;  // 命中的排名（1-6），未命中为-1
}

export function validate(
  prediction: PredictionResult,
  actual: Period
): ValidationResult {
  const index = prediction.candidates.indexOf(actual.shengxiao);
  return {
    period: prediction.period,
    actual: actual.shengxiao,
    predicted: prediction.candidates,
    hit: index !== -1,
    rank: index === -1 ? -1 : index + 1
  };
}

export function batchValidate(
  historyData: Period[],
  startPeriod: number
): ValidationResult[] {
  const results: ValidationResult[] = [];
  
  for (let i = startPeriod; i < historyData.length; i++) {
    const currentPeriod = historyData[i].period;
    try {
      const prediction = predict(historyData, currentPeriod);
      const actual = historyData[i];
      results.push(validate(prediction, actual));
    } catch (e) {
      console.error(`Failed to predict period ${currentPeriod}:`, e);
    }
  }
  
  return results;
}

export function calculateHitRate(results: ValidationResult[]): {
  totalHit: number;
  hitRate: number;
  firstHit: number;
  firstHitRate: number;
  rankDistribution: Record<number, number>;
} {
  const totalHit = results.filter(r => r.hit).length;
  const firstHit = results.filter(r => r.rank === 1).length;
  
  const rankDistribution: Record<number, number> = {};
  results.forEach(r => {
    if (r.rank > 0) {
      rankDistribution[r.rank] = (rankDistribution[r.rank] || 0) + 1;
    }
  });
  
  return {
    totalHit,
    hitRate: totalHit / results.length,
    firstHit,
    firstHitRate: firstHit / results.length,
    rankDistribution
  };
}
```

```typescript
          shengxiao: shengxiao as Shengxiao,

    reader.readAsText(file);
    return false;
```

### 6.1 Zustand Store

```typescript
// src/store/useAppStore.ts

import { create } from 'zustand';
import { Period, PredictionResult } from '../algorithm';

interface AppState {
  historyData: Period[];
  prediction: PredictionResult | null;
  validationResults: ValidationResult[];
  
  // Actions
  setHistoryData: (data: Period[]) => void;
  setPrediction: (prediction: PredictionResult | null) => void;
  setValidationResults: (results: ValidationResult[]) => void;
  
  // Computed
  getCurrentPeriod: () => number;
  getLastResult: () => Period | null;
}

export const useAppStore = create<AppState>((set, get) => ({
  historyData: [],
  prediction: null,
  validationResults: [],
  
  setHistoryData: (data) => set({ historyData: data }),
  setPrediction: (prediction) => set({ prediction }),
  setValidationResults: (results) => set({ validationResults: results }),
  
  getCurrentPeriod: () => {
    const { historyData } = get();
    if (historyData.length === 0) return 0;
    return Math.max(...historyData.map(p => p.period)) + 1;
  },
  
  getLastResult: () => {
    const { historyData } = get();
    if (historyData.length === 0) return null;
    return historyData[historyData.length - 1];
  }
}));
```
